Automatización con Python


def procesar_datos(archivo):
    """
    Procesa datos automáticamente.
    """
    with open(archivo, 'r') as f:
        datos = f.read()

    # Procesar datos
    resultado = datos.strip().upper()

    return resultado

# Uso simple
resultado = procesar_datos('input.txt')
print(f"Procesado: {resultado}")
