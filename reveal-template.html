import os
from pathlib import Path

def organizar_archivos(directorio):
    """Organiza archivos por extensi√≥n."""
    base_path = Path(directorio)

    for archivo in base_path.glob('*.*'):
        extension = archivo.suffix[1:]  # Sin el punto
        carpeta_destino = base_path / extension
        carpeta_destino.mkdir(exist_ok=True)
        archivo.rename(carpeta_destino / archivo.name)

# Uso

from collections import Counter
import json

def analizar_logs(archivo_log):
    """Analiza logs y extrae estad√≠sticas."""
    with open(archivo_log, 'r') as f:
        lineas = f.readlines()

    # Contar c√≥digos de estado
    codigos = [linea.split()[8] for linea in lineas
               if len(linea.split()) > 8]
    contador = Counter(codigos)

    return dict(contador)

# Uso
